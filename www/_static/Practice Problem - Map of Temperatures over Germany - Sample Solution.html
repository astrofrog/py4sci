import json
import requests

BASE_URL = 'http://api.openweathermap.org/data/2.5/weather'

def get_temperature(lon, lat):
    data = requests.get(BASE_URL, params={'lon': lon, 'lat':lat}).json()
    return data[u'main'][u'temp']
